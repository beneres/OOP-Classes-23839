using System;namespace amountOfTime{	class Program	{		static void Main(string[] args)		{			Console.WriteLine("Enter the total amount of time in minutes only");			int min = int.Parse(Console.ReadLine());			TimeSpan elapsedTime = new TimeSpan(0, min, 0);			int hour = elapsedTime.Hours;			int remainingMinutes = elapsedTime.Minutes;			Console.WriteLine(string.Format("{0} minutes is equal to {1} hours and {2} minutes", min, hour, remainingMinutes));					}	}}
