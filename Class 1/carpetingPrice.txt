using System;namespace carpetingPrice{    class Program    {        static void Main(string[] args)        {            double length, width;            const double per_square_foot = 13.20;            Console.WriteLine("Enter length of room in feet: ");            length = double.Parse(Console.ReadLine());            Console.WriteLine("Enter width of room in feet:");            width = double.Parse(Console.ReadLine());            double area = length * width;            double price = area * per_square_foot;            Console.WriteLine("The cost to carpet this {0} by {1} room is ${2}", length, width, price);            Console.ReadLine();        }    }}
