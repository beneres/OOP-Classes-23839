using System;namespace HourlyPayRate{    class Program    {        static void Main(string[] args)        {            double minPayRate = 5.65;            double maxPayRate = 49.99;            string inputString;            double userInput;            do            {             Console.WriteLine("Please enter your hourly pay rate");             inputString = Console.ReadLine();             userInput = Convert.ToDouble(inputString);            }            while (userInput < minPayRate || userInput > maxPayRate);            Console.WriteLine("The pay rate is {0}", userInput);        }            }}
