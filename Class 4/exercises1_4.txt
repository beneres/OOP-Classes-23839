using System;namespace exercise1_4{    class Program    {        static void Main(string[] args)        {            //Create a console - based application whose Main() method asks the user to input an integer and then calls            //a method named MultiplicationTable(), which displays the results of multiplying the integer by each of the            //numbers 2 through 10                     {                Console.Write("Please enter an integer : ");                string input = Console.ReadLine();                int num;                int.TryParse(input, out num); // returns num = 0 if input invalid                Console.WriteLine();                multiplicationTable(num);                Console.ReadKey();            }            static void multiplicationTable(int num)            {                for (int i = 2; i <= 10; i++)                {                    Console.WriteLine("{0} x {1, -2} = {2}", num, i, num * i);                }            }        }    }}
